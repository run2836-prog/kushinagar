import turtle import time import random # Screen setup screen = turtle.Screen() screen.title("BANANA GAME - Saare Options Ke Saath") screen.bgcolor("lightblue") screen.setup(width=800, height=600) # Score variable score = 0 high_score = 0 # Score display score_display = turtle.Turtle() score_display.speed(0) score_display.color("black") score_display.penup() score_display.hideturtle() score_display.goto(0, 260) score_display.write(f"Score: {score} High Score: {high_score}", align="center", font=("Arial", 24, "normal")) # Player (Banana Wala) player = turtle.Turtle() player.shape("turtle") # Turtle ko banana samjho player.color("yellow") player.penup() player.speed(0) player.goto(0, -250) player.setheading(90) # Up facing # Player speed player_speed = 20 # Bananas list bananas = [] # Special bananas special_bananas = [] # Falling objects (rocks) rocks = [] # Power-ups powerups = [] # Game variables game_active = True game_paused = False level = 1 banana_speed = 3 banana_count = 5 lives = 3 # Lives display lives_display = turtle.Turtle() lives_display.speed(0) lives_display.color("red") lives_display.penup() lives_display.hideturtle() lives_display.goto(-350, 260) lives_display.write(f"Lives: {lives}", align="left", font=("Arial", 18, "normal")) # Level display level_display = turtle.Turtle() level_display.speed(0) level_display.color("purple") level_display.penup() level_display.hideturtle() level_display.goto(350, 260) level_display.write(f"Level: {level}", align="right", font=("Arial", 18, "normal")) # Functions for movement def move_left(): x = player.xcor() if x > -380: x -= player_speed player.setx(x) def move_right(): x = player.xcor() if x < 380: x += player_speed player.setx(x) def move_up(): y = player.ycor() if y < 250: y += player_speed player.sety(y) def move_down(): y = player.ycor() if y > -250: y -= player_speed player.sety(y) def pause_game(): global game_paused game_paused = not game_paused if game_paused: pause_message.showturtle() else: pause_message.hideturtle() def restart_game(): global game_active, score, lives, level, banana_speed, game_paused game_active = True game_paused = False score = 0 lives = 3 level = 1 banana_speed = 3 # Clear all objects for banana in bananas: banana.hideturtle() for rock in rocks: rock.hideturtle() for special in special_bananas: special.hideturtle() for powerup in powerups: powerup.hideturtle() bananas.clear() rocks.clear() special_bananas.clear() powerups.clear() player.goto(0, -250) pause_message.hideturtle() update_score() update_lives() update_level() def quit_game(): global game_active game_active = False screen.bye() def update_score(): score_display.clear() score_display.write(f"Score: {score} High Score: {high_score}", align="center", font=("Arial", 24, "normal")) def update_lives(): lives_display.clear() lives_display.write(f"Lives: {lives}", align="left", font=("Arial", 18, "normal")) def update_level(): level_display.clear() level_display.write(f"Level: {level}", align="right", font=("Arial", 18, "normal")) def create_banana(): if len(bananas) < banana_count and game_active and not game_paused: banana = turtle.Turtle() banana.shape("circle") banana.color("yellow") banana.penup() banana.speed(0) x = random.randint(-380, 380) banana.goto(x, 300) banana.dy = banana_speed bananas.append(banana) # Schedule next banana screen.ontimer(create_banana, random.randint(500, 1500)) def create_special_banana(): if game_active and not game_paused and random.random() < 0.3: # 30% chance special = turtle.Turtle() special.shape("circle") special.color("gold") special.penup() special.speed(0) x = random.randint(-380, 380) special.goto(x, 300) special.dy = banana_speed * 1.5 special_bananas.append(special) screen.ontimer(create_special_banana, random.randint(2000, 5000)) def create_rock(): if game_active and not game_paused and random.random() < 0.2: # 20% chance rock = turtle.Turtle() rock.shape("square") rock.color("gray") rock.penup() rock.speed(0) rock.shapesize(stretch_wid=1, stretch_len=2) x = random.randint(-380, 380) rock.goto(x, 300) rock.dy = banana_speed * 2 rocks.append(rock) screen.ontimer(create_rock, random.randint(3000, 6000)) def create_powerup(): if game_active and not game_paused and random.random() < 0.1: # 10% chance powerup = turtle.Turtle() powerup.shape("triangle") powerup.color("green") powerup.penup() powerup.speed(0) x = random.randint(-380, 380) powerup.goto(x, 300) powerup.dy = banana_speed powerups.append(powerup) screen.ontimer(create_powerup, random.randint(5000, 10000)) # Keyboard bindings screen.listen() screen.onkeypress(move_left, "Left") screen.onkeypress(move_right, "Right") screen.onkeypress(move_up, "Up") screen.onkeypress(move_down, "Down") screen.onkeypress(pause_game, "p") screen.onkeypress(pause_game, "P") screen.onkeypress(restart_game, "r") screen.onkeypress(restart_game, "R") screen.onkeypress(quit_game, "q") screen.onkeypress(quit_game, "Q") # Instructions instructions = turtle.Turtle() instructions.speed(0) instructions.color("black") instructions.penup() instructions.hideturtle() instructions.goto(0, -300) instructions.write("Arrow Keys: Move | P: Pause | R: Restart | Q: Quit", align="center", font=("Arial", 14, "normal")) # Pause message pause_message = turtle.Turtle() pause_message.speed(0) pause_message.color("red") pause_message.penup() pause_message.hideturtle() pause_message.goto(0, 0) pause_message.write("GAME PAUSED", align="center", font=("Arial", 36, "bold")) # Start creating objects create_banana() create_special_banana() create_rock() create_powerup() # Game loop while game_active: if not game_paused: # Move bananas for banana in bananas[:]: y = banana.ycor() y -= banana.dy banana.sety(y) # Check collision with player if banana.distance(player) < 30: banana.hideturtle() bananas.remove(banana) score += 10 if score > high_score: high_score = score update_score() # Level up based on score if score // 100 >= level: level += 1 banana_speed += 0.5 banana_count += 2 update_level() # Remove if off screen elif y < -300: banana.hideturtle() bananas.remove(banana) # Move special bananas for special in special_bananas[:]: y = special.ycor() y -= special.dy special.sety(y) if special.distance(player) < 30: special.hideturtle() special_bananas.remove(special) score += 50 if score > high_score: high_score = score update_score() elif y < -300: special.hideturtle() special_bananas.remove(special) # Move rocks for rock in rocks[:]: y = rock.ycor() y -= rock.dy rock.sety(y) if rock.distance(player) < 40: rock.hideturtle() rocks.remove(rock) lives -= 1 update_lives() if lives <= 0: game_active = False game_over_message = turtle.Turtle() game_over_message.color("red") game_over_message.penup() game_over_message.hideturtle() game_over_message.goto(0, 0) game_over_message.write(f"GAME OVER! Final Score: {score}", align="center", font=("Arial", 30, "bold")) time.sleep(2) elif y < -300: rock.hideturtle() rocks.remove(rock) # Move powerups for powerup in powerups[:]: y = powerup.ycor() y -= powerup.dy powerup.sety(y) if powerup.distance(player) < 30: powerup.hideturtle() powerups.remove(powerup) lives += 1 update_lives() elif y < -300: powerup.hideturtle() powerups.remove(powerup) screen.update() time.sleep(0.01) screen.mainloop()
