import turtle import time import random import math # Screen setup screen = turtle.Screen() screen.title("🏁 REAL CAR RACING GAME 🏁") screen.bgcolor("black") screen.setup(width=800, height=600) screen.tracer(0) # Score and game variables score = 0 high_score = 0 level = 1 lives = 3 game_active = True game_paused = False speed = 5 car_width = 40 car_height = 60 # Score display score_display = turtle.Turtle() score_display.speed(0) score_display.color("white") score_display.penup() score_display.hideturtle() score_display.goto(0, 260) score_display.write(f"Score: {score} High Score: {high_score}", align="center", font=("Arial", 24, "bold")) # Lives display lives_display = turtle.Turtle() lives_display.speed(0) lives_display.color("red") lives_display.penup() lives_display.hideturtle() lives_display.goto(-350, 260) lives_display.write(f"Lives: {lives} ❤️", align="left", font=("Arial", 18, "bold")) # Level display level_display = turtle.Turtle() level_display.speed(0) level_display.color("yellow") level_display.penup() level_display.hideturtle() level_display.goto(350, 260) level_display.write(f"Level: {level}", align="right", font=("Arial", 18, "bold")) # Road lines for animation road_lines = [] for i in range(8): line = turtle.Turtle() line.shape("square") line.color("white") line.shapesize(stretch_wid=1, stretch_len=0.5) line.penup() line.goto(0, -300 + i * 100) line.speed(0) road_lines.append(line) # Player's Car (Realistic design) player = turtle.Turtle() player.shape("square") player.color("red") player.shapesize(stretch_wid=1.5, stretch_len=2.5) # Car shape player.penup() player.speed(0) player.goto(0, -250) # Add car details (wheels) wheel1 = turtle.Turtle() wheel1.shape("circle") wheel1.color("black") wheel1.shapesize(stretch_wid=0.5, stretch_len=0.5) wheel1.penup() wheel1.goto(-25, -265) wheel2 = turtle.Turtle() wheel2.shape("circle") wheel2.color("black") wheel2.shapesize(stretch_wid=0.5, stretch_len=0.5) wheel2.penup() wheel2.goto(25, -265) wheel3 = turtle.Turtle() wheel3.shape("circle") wheel3.color("black") wheel3.shapesize(stretch_wid=0.5, stretch_len=0.5) wheel3.penup() wheel3.goto(-25, -235) wheel4 = turtle.Turtle() wheel4.shape("circle") wheel4.color("black") wheel4.shapesize(stretch_wid=0.5, stretch_len=0.5) wheel4.penup() wheel4.goto(25, -235) # Opponent cars list opponents = [] # Power-ups list powerups = [] # Coin list for collection coins = [] # Car colors car_colors = ["blue", "green", "orange", "purple", "pink", "cyan", "yellow", "white"] # Movement functions def move_left(): x = player.xcor() if x > -320: # Left boundary x -= 30 player.setx(x) update_wheels() def move_right(): x = player.xcor() if x < 320: # Right boundary x += 30 player.setx(x) update_wheels() def move_up(): y = player.ycor() if y < 200: y += 30 player.sety(y) update_wheels() def move_down(): y = player.ycor() if y > -280: y -= 30 player.sety(y) update_wheels() def update_wheels(): # Update wheel positions wheel1.goto(player.xcor() - 25, player.ycor() - 15) wheel2.goto(player.xcor() + 25, player.ycor() - 15) wheel3.goto(player.xcor() - 25, player.ycor() + 15) wheel4.goto(player.xcor() + 25, player.ycor() + 15) def pause_game(): global game_paused game_paused = not game_paused if game_paused: pause_message.showturtle() else: pause_message.hideturtle() def restart_game(): global game_active, score, lives, level, speed, game_paused game_active = True game_paused = False score = 0 lives = 3 level = 1 speed = 5 # Clear all objects for opponent in opponents: opponent.hideturtle() for powerup in powerups: powerup.hideturtle() for coin in coins: coin.hideturtle() opponents.clear() powerups.clear() coins.clear() player.goto(0, -250) update_wheels() pause_message.hideturtle() update_display() def quit_game(): global game_active game_active = False screen.bye() def update_display(): global high_score if score > high_score: high_score = score score_display.clear() score_display.write(f"Score: {score} High Score: {high_score}", align="center", font=("Arial", 24, "bold")) lives_display.clear() lives_display.write(f"Lives: {lives} ❤️", align="left", font=("Arial", 18, "bold")) level_display.clear() level_display.write(f"Level: {level}", align="right", font=("Arial", 18, "bold")) def create_opponent(): if len(opponents) < 5 + level and game_active and not game_paused: opponent = turtle.Turtle() opponent.shape("square") opponent.color(random.choice(car_colors)) opponent.shapesize(stretch_wid=1.5, stretch_len=2.5) opponent.penup() opponent.speed(0) # Random position x = random.randint(-300, 300) y = random.randint(200, 400) opponent.goto(x, y) # Store additional data opponent.speed = random.randint(3, 8) + level opponent.direction = random.choice([-1, 1]) # Lane changing opponent.lane_change_timer = 0 opponents.append(opponent) # Schedule next opponent screen.ontimer(create_opponent, random.randint(1000, 3000)) def create_powerup(): if game_active and not game_paused and random.random() < 0.2: powerup = turtle.Turtle() powerup.shape("circle") powerup.color("green") powerup.shapesize(stretch_wid=0.8, stretch_len=0.8) powerup.penup() powerup.speed(0) x = random.randint(-300, 300) y = random.randint(300, 400) powerup.goto(x, y) powerup.type = random.choice(["speed", "shield", "points"]) powerup.color("lightgreen" if powerup.type == "speed" else "gold" if powerup.type == "points" else "cyan") powerups.append(powerup) screen.ontimer(create_powerup, random.randint(5000, 10000)) def create_coin(): if game_active and not game_paused and random.random() < 0.3: coin = turtle.Turtle() coin.shape("circle") coin.color("yellow") coin.shapesize(stretch_wid=0.5, stretch_len=0.5) coin.penup() coin.speed(0) x = random.randint(-300, 300) y = random.randint(300, 400) coin.goto(x, y) coin.value = random.randint(5, 20) coins.append(coin) screen.ontimer(create_coin, random.randint(2000, 4000)) def move_road(): # Animate road lines for line in road_lines: y = line.ycor() y -= speed if y < -300: y = 300 line.sety(y) def check_collision(): global lives, game_active, score # Check opponent collisions for opponent in opponents[:]: distance = math.sqrt((player.xcor() - opponent.xcor())**2 + (player.ycor() - opponent.ycor())**2) if distance < 40: # Collision detected opponent.hideturtle() opponents.remove(opponent) lives -= 1 update_display() # Flash red on collision player.color("darkred") screen.update() time.sleep(0.1) player.color("red") if lives <= 0: game_active = False game_over() # Check coin collection for coin in coins[:]: if player.distance(coin) < 30: coin.hideturtle() coins.remove(coin) score += coin.value update_display() # Check powerup collection for powerup in powerups[:]: if player.distance(powerup) < 30: powerup.hideturtle() powerups.remove(powerup) if powerup.type == "points": score += 100 elif powerup.type == "speed": global speed speed += 2 screen.ontimer(lambda: restore_speed(), 5000) elif powerup.type == "shield": lives += 1 update_display() def restore_speed(): global speed speed = 5 + level def game_over(): game_over_msg = turtle.Turtle() game_over_msg.color("red") game_over_msg.penup() game_over_msg.hideturtle() game_over_msg.goto(0, 0) game_over_msg.write(f"GAME OVER!\nFinal Score: {score}\nPress R to Restart", align="center", font=("Arial", 30, "bold")) # Keyboard bindings screen.listen() screen.onkeypress(move_left, "Left") screen.onkeypress(move_right, "Right") screen.onkeypress(move_up, "Up") screen.onkeypress(move_down, "Down") screen.onkeypress(pause_game, "p") screen.onkeypress(pause_game, "P") screen.onkeypress(restart_game, "r") screen.onkeypress(restart_game, "R") screen.onkeypress(quit_game, "q") screen.onkeypress(quit_game, "Q") # Instructions instructions = turtle.Turtle() instructions.speed(0) instructions.color("gray") instructions.penup() instructions.hideturtle() instructions.goto(0, -320) instructions.write("⬅️➡️⬆️⬇️: Move | P: Pause | R: Restart | Q: Quit", align="center", font=("Arial", 14, "normal")) # Pause message pause_message = turtle.Turtle() pause_message.speed(0) pause_message.color("yellow") pause_message.penup() pause_message.hideturtle() pause_message.goto(0, 0) pause_message.write("⏸️ GAME PAUSED ⏸️", align="center", font=("Arial", 36, "bold")) # Road borders left_border = turtle.Turtle() left_border.color("white") left_border.penup() left_border.goto(-350, 300) left_border.pendown() left_border.goto(-350, -300) right_border = turtle.Turtle() right_border.color("white") right_border.penup() right_border.goto(350, 300) right_border.pendown() right_border.goto(350, -300) # Start creating objects create_opponent() create_powerup() create_coin() # Game loop while game_active: if not game_paused: # Move road move_road() # Move opponents for opponent in opponents[:]: # Move down y = opponent.ycor() y -= opponent.speed opponent.sety(y) # Random lane changing opponent.lane_change_timer += 1 if opponent.lane_change_timer > 30: opponent.direction = random.choice([-1, 0, 1]) opponent.lane_change_timer = 0 # Move sideways x = opponent.xcor() x += opponent.direction * 2 if -300 < x < 300: opponent.setx(x) # Remove if off screen if y < -300: opponent.hideturtle() opponents.remove(opponent) score += 20 # Bonus for avoiding update_display() # Move powerups for powerup in powerups[:]: y = powerup.ycor() y -= speed powerup.sety(y) if y < -300: powerup.hideturtle() powerups.remove(powerup) # Move coins for coin in coins[:]: y = coin.ycor() y -= speed coin.sety(y) if y < -300: coin.hideturtle() coins.remove(coin) # Check collisions check_collision() # Level up system if score > level * 200: level += 1 speed += 1 update_display() screen.update() time.sleep(0.01) screen.mainloop()
